## 方法一

层序遍历相信大家都知道，现在的问题是如何输出当前的层级，我们可以通过添加last和nlast这两个变量来实现：

1. last表示当前层级的最右结点
2. nlast是当前层级下一层级的最右结点

如何当前层级下一层级的最右结点呢？

可以每次将二叉树中的结点node放入queue中时，将该node赋给nlast，**这样当前层级的最右结点出栈后，nlast表示就是当前层级下一层级的最右结点**。

如果用cur表示当前出栈的结点，当`cur==last && !queue.isEmpty()`时，就表示已到达当前层级的最右结点，让`last=nlast`即可。



```java
import java.util.Queue;
import java.util.LinkedList;
import java.util.ArrayList;
import java.util.Scanner;

public class Main{
    
    static class Node{
        int value;
        Node left,right;
        public Node(int value){
            this.value=value;
        }
    }
    
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int root=sc.nextInt();
        Node[] nodes=new Node[n+1];
        Node node;
        for(int i=1;i<=n;i++){
            node=new Node(i);
            nodes[i]=node;
        }
        int value,left,right;
        for(int i=1;i<=n;i++){
            value=sc.nextInt();
            left=sc.nextInt();
            right=sc.nextInt();
            nodes[value].left=left==0?null:nodes[left];
            nodes[value].right=right==0?null:nodes[right];
            
        }
        Node head=nodes[root];
        levelPint(head);
        System.out.println();
        ZigZagPint(head);
    }
    
    public static void levelPint(Node head){
        if(head==null) return;
        Queue<Node> queue=new LinkedList<Node>();
        queue.offer(head);
        int level=1;
        System.out.print("Level "+level+" : ");
        Node cur;
        //last是当前层级的最右结点，nlast是当前层级下一层级的最右结点
        Node last=head,nlast=null;
        while(!queue.isEmpty()){
            cur=queue.poll();
            System.out.print(cur.value+" ");
            if(cur.left!=null){
                queue.offer(cur.left);
                nlast=cur.left;
            }
            if(cur.right!=null){
                queue.offer(cur.right);
                nlast=cur.right;
            }
            if(last==cur&&!queue.isEmpty()){
                level++;
                last=nlast;
                System.out.println();
                System.out.print("Level "+level+" : ");
            }
            
        }
        
    }
    
    //当从左到右打印时直接打印即可
    //从右到做打印时，先把数存起来，再逆序打印
    public static void ZigZagPint(Node head){
        if(head==null) return;
        Queue<Node> queue=new LinkedList<Node>();
        ArrayList<Integer> list=new ArrayList<Integer>();//存放逆序输出的值
        queue.offer(head);
        int level=1;
        System.out.print("Level "+level+" from left to right: ");
        Node cur;
        //last是当前层级的最右结点，nlast是当前层级下一层级的最右结点
        Node last=head,nlast=null;
        while(!queue.isEmpty()){
            cur=queue.poll();
            if(level%2==0){//逆序通过list输出
                list.add(cur.value);
            }else{//正序直接输出
                System.out.print(cur.value+" ");
            }
            if(cur.left!=null){
                queue.offer(cur.left);
                nlast=cur.left;
            }
            if(cur.right!=null){
                queue.offer(cur.right);
                nlast=cur.right;
            }
            if(last==cur&&!queue.isEmpty()){
                
                if(level%2==0){//输出上一次的逆序
                    printReverse(list);
                    list.clear();//清空list
                }
                
                level++;
                last=nlast;
                System.out.println();
                if(level%2==0){//逆序前缀
                    System.out.print("Level "+level+" from right to left: ");
                }else{//正序前缀
                    System.out.print("Level "+level+" from left to right: ");
                }
                
            }

        }
        //输出最后一层的逆序值
        if(level%2==0&&!list.isEmpty()){
            printReverse(list);
        }

    }
    //逆序输出
    public static void printReverse(ArrayList<Integer> list){
        for(int i=list.size()-1;i>=0;i--){
            System.out.print(list.get(i)+" ");
        }
    }
}
```



## 方法二（推荐）

通过`null`结点对每一层的结尾做标记，这样就可以知道什么时候到了每层的结尾了。



```java
import java.util.Queue;
import java.util.LinkedList;
import java.util.ArrayList;
import java.util.Scanner;

public class Main{
    
    static class Node{
        int value;
        Node left,right;
        public Node(int value){
            this.value=value;
        }
    }
    
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        int root=sc.nextInt();
        Node[] nodes=new Node[n+1];
        Node node;
        for(int i=1;i<=n;i++){
            node=new Node(i);
            nodes[i]=node;
        }
        int value,left,right;
        for(int i=1;i<=n;i++){
            value=sc.nextInt();
            left=sc.nextInt();
            right=sc.nextInt();
            nodes[value].left=left==0?null:nodes[left];
            nodes[value].right=right==0?null:nodes[right];
            
        }
        Node head=nodes[root];
        levelPrint(head);
        System.out.println();
        zigzagPrint(head);
    }

    public static void levelPrint(Node head){
        if(head==null) return;
        Queue<Node> queue=new LinkedList<>();
        queue.offer(head);
        queue.offer(null);
        int index=1;
        System.out.print("Level "+index+" : ");
        Node cur;
        while(!queue.isEmpty()){
            cur=queue.poll();
            if(cur==null){
                if(queue.isEmpty()) break;
                System.out.println();
                index++;
                System.out.print("Level "+index+" : ");
                queue.offer(null);//只要该层的结尾出栈，下一层的所有节点就已经入栈了，所以此时入栈的null是下一层的结尾
                continue;
            }else System.out.print(cur.value+" ");
            if(cur.left!=null) queue.offer(cur.left);
            if(cur.right!=null) queue.offer(cur.right);
        }
    }
    
    public static void zigzagPrint(Node head){
        if(head==null) return;
        Queue<Node> queue=new LinkedList<>();
        queue.offer(head);
        queue.offer(null);
        int index=1;
        System.out.print("Level "+index+" from left to right: ");
        ArrayList<Integer> list=new ArrayList<Integer>();
        Node cur;
        while(!queue.isEmpty()){
            cur=queue.poll();
            if(cur==null){
                //这里若果index是奇数，说明已经打印完了
                //如果是偶数，则打印的数放在了list中
                if(index%2==0){
                    for(int i=list.size()-1;i>=0;i--){
                        System.out.print(list.get(i)+" ");
                    }
                    list.clear();
                }
                
                if(queue.isEmpty()) break;
                
                System.out.println();
                index++;
                if(index%2==0) System.out.print("Level "+index+" from right to left: ");
                else System.out.print("Level "+index+" from left to right: ");
                queue.offer(null);
                continue;
            }else {
                if(index%2==0) list.add(cur.value);
                else System.out.print(cur.value+" ");
            }
            if(cur.left!=null) queue.offer(cur.left);
            if(cur.right!=null) queue.offer(cur.right);
        }
        
        
    }
}
```

```java

        int index=1;
        
        while(!queue.isEmpty()){
            int sz=queue.size();

                Node cur=queue.poll();
                System.out.print(cur.value+" ");
                if(cur.left!=null) queue.offer(cur.left);
                if(cur.right!=null) queue.offer(cur.right);
            }
            System.out.println();
            index++;
        }
    }
```
